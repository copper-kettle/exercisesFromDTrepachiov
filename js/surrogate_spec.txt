random_letter.js
export arr_letter(sourse, arr_lenght, arg), где:
sourse - строка не менее 20 слов,
arr_lenght - целое число не более 20.
arg == "pst": (возвращает массив длины arr_lenght,
составленный из слов, содержащихся в sourse c добавлением,
в случайном порядке постфикса ".html")

arg == "prf": (возвращает массив длины arr_lenght,
составленный из слов, содержащихся в sourse c добавлением,
в случайном порядке префикса "http//")
